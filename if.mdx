For example:

我们做一个判断**闰年**的计算器。

> **闰年逻辑**：标准公历规则——能被4整除且**不能被100整除，或能被400整除**。

```go
package main

import "fmt"

func main() {
    var input string
    fmt.Print("请输入年份：")
    // 读取一行输入（含空格等），保存为字符串
    fmt.Scanln(&input)

    var year int
    // 使用 fmt.Sscanf 从字符串中解析出年份
    n, err := fmt.Sscanf(input, "%d", &year)
    if err != nil || n != 1 {
        fmt.Println("输入无效，请输入一个整数。")
        return
    }

    // 闰年判定：能被4整除且不能被100整除，或能被400整除
    if (year%4 == 0 && year%100 != 0) || year%400 == 0 {
        fmt.Printf("%d 年是闰年。\n", year)
    } else {
        fmt.Printf("%d 年是平年（普通年）。\n", year)
    }
}
```
